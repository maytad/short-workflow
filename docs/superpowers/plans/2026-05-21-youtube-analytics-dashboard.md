# YouTube Analytics Dashboard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the approved YouTube analytics dashboard with recent API refresh, linked and unlinked videos, rule-based diagnosis, and on-demand `gpt-5.5` AI diagnosis using reasoning effort `xhigh`.

**Architecture:** Add a small analytics vertical across `packages/shared`, `packages/db`, `packages/ai`, `apps/api`, and `apps/web`. The API owns YouTube and OpenAI orchestration, the database stores video links, metric snapshots, and diagnoses, and the web app renders a dense operational dashboard without exposing secrets.

**Tech Stack:** Bun workspaces, Turborepo, ElysiaJS, Drizzle ORM, hosted Supabase Postgres, React + Vite + TanStack Router + TanStack Query, Tailwind v4, lucide-react, OpenAI Responses API, YouTube Data API, YouTube Analytics API.

## Source Spec

Implement from:

`docs/superpowers/specs/2026-05-21-youtube-analytics-dashboard-design.md`

Approved refinement:

- Redesign `/analytics` as a decision-first public-only dashboard.
- Keep All uploads available for private/scheduled troubleshooting, but do not let those rows drive headline public performance decisions.
- Add Needs action, Winners, New, All public, and All uploads review queues.
- Replace raw-first detail content with What happened, Likely cause, Next action, and Prompt & title fixes.
- Note Shown in feed and How many chose to view as future manual or CSV-import metrics.
- Tighten Tiny Mechanisms prompt rules around first-second visible contradiction, object-plus-surprising-behavior titles, and action-first image prompts.

Keep these scope boundaries:

- No CSV import in this version.
- No scheduled sync or worker analytics job.
- No batch AI diagnosis.
- No automatic prompt, seed, scene, or upload mutation from diagnosis.
- Refresh is manual and defaults to the last 30 days.
- AI diagnosis is on-demand only and uses `gpt-5.5` with reasoning effort `xhigh`.

create table youtube_video_links (
  id uuid primary key default gen_random_uuid(),

  project_id uuid references projects(id) on delete set null,
  upload_job_id uuid references jobs(id) on delete set null,
  source text not null,
  link_status text not null,
  title text not null,
  description text,
  published_at timestamptz,
  duration_seconds integer,
  privacy_status text,
  last_synced_at timestamptz,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  constraint youtube_video_links_source_check check (source in ('db_upload', 'channel_discovery')),
  constraint youtube_video_links_status_check check (link_status in ('linked', 'unlinked'))
);

create unique index youtube_video_links_video_id_idx
on youtube_video_links (youtube_video_id);

create index youtube_video_links_project_idx
on youtube_video_links (project_id);

create index youtube_video_links_published_at_idx
on youtube_video_links (published_at desc);

create table youtube_analytics_snapshots (
  id uuid primary key default gen_random_uuid(),
  youtube_video_link_id uuid not null references youtube_video_links(id) on delete cascade,
  youtube_video_id text not null,
  snapshot_at timestamptz not null default now(),
  window_days integer not null,
  views integer,
  engaged_views integer,
  likes integer,
  comments integer,
  shares integer,
  subscribers_gained integer,
  average_view_duration_seconds integer,
  average_view_percentage numeric,

);

create index youtube_analytics_snapshots_link_time_idx

create index youtube_analytics_snapshots_video_time_idx

create table youtube_video_diagnoses (

async function writeYoutubeToken(env: Env, token: YoutubeToken) {
  await mkdir(youtubeDir(env.LOCAL_ASSET_ROOT), { recursive: true, mode: 0o700 });
  await writeFile(youtubeTokenPath(env.LOCAL_ASSET_ROOT), JSON.stringify(token, null, 2), {

export async function refreshYoutubeToken(input: {
  env?: Env;
  token: YoutubeToken;
  fetchFn?: typeof fetch;
  now?: number;
}) {
  const env = input.env ?? parseEnv();
  if (!env.YOUTUBE_OAUTH_CLIENT_ID) {
    throw new Error("youtube_oauth_not_configured");
  }

  const body = new URLSearchParams({
    client_id: env.YOUTUBE_OAUTH_CLIENT_ID,
    grant_type: "refresh_token",
    refresh_token: input.token.refresh_token,
  });

  if (env.YOUTUBE_OAUTH_CLIENT_SECRET) {
    body.set("client_secret", env.YOUTUBE_OAUTH_CLIENT_SECRET);
  }

  const response = await (input.fetchFn ?? fetch)("https://oauth2.googleapis.com/token", {
    body,
    headers: {
      "content-type": "application/x-www-form-urlencoded",
    },
    method: "POST",
  });

  if (!response.ok) {
    throw new Error("youtube_token_refresh_failed");
  }

  const parsed = youtubeRefreshTokenResponseSchema.parse(await response.json());
  const now = input.now ?? Date.now();
  return storedTokenSchema.parse({
    ...input.token,
    ...parsed,
    refresh_token: input.token.refresh_token,
    expires_at: new Date(now + (parsed.expires_in ?? 3600) * 1000).toISOString(),
  });
}

export async function readFreshYoutubeAccessToken(input: {
  env?: Env;
  fetchFn?: typeof fetch;
  now?: number;
} = {}) {
  const env = input.env ?? parseEnv();
  let token = await readYoutubeToken(env);

  if (!hasRequiredYoutubeScopes(token.scope)) {
    throw new Error("youtube_reconnect_required");
  }

  if (isYoutubeTokenExpired(token, input.now)) {
    token = await refreshYoutubeToken({ env, token, fetchFn: input.fetchFn, now: input.now });
    await writeYoutubeToken(env, token);
  }

  return token.access_token;
}

export function parseIso8601DurationSeconds(value: string) {
  const match = /^PT(?:(\d+)H)?(?:(\d+)M)?(?:(\d+)S)?$/.exec(value);
  if (!match) {
    return null;
  }

  const hours = Number(match[1] ?? 0);
  const minutes = Number(match[2] ?? 0);
  const seconds = Number(match[3] ?? 0);
  return hours * 3600 + minutes * 60 + seconds;
}

export function deriveSnapshotMetrics(input: DerivedMetricInput) {
  const ageHours =
    input.publishedAt === null
      ? null
      : Math.max(0, (input.now.getTime() - input.publishedAt.getTime()) / 3_600_000);
  const viewsPerHour =
    input.views !== null && ageHours !== null && ageHours > 0 ? input.views / ageHours : null;
  const likeRate =
    input.views !== null && input.likes !== null && input.views > 0
      ? (input.likes / input.views) * 100
      : null;

  return {
    ageHours,
    viewsPerHour,
    likeRate,
  };

export function buildRuleDiagnosis(input: RuleDiagnosisInput): RuleDiagnosisOutput {
  if (input.ageHours !== null && input.ageHours < 3) {
    return {
      labels: ["too_new"],
      priority: "low",
      summaryTh: "คลิปยังใหม่เกินไปสำหรับการสรุป performance อย่างมั่นใจ",
      suggestionsEn: { labels: ["too_new"], priority: "low" },
    };
  }

  const labels: string[] = [];
  const views = input.snapshot.views ?? 0;
  const viewsPerHour = input.snapshot.viewsPerHour ?? 0;
  const medianViews = input.recentMedians.views ?? 0;
  const medianViewsPerHour = input.recentMedians.viewsPerHour ?? 0;
  const retention = input.snapshot.averageViewPercentage;
  const medianRetention = input.recentMedians.averageViewPercentage;
  const likeRate = input.snapshot.likeRate;
  const medianLikeRate = input.recentMedians.likeRate;

  if (medianViews > 0 && views < medianViews * 0.35) {
    labels.push("low_exposure_proxy");
  }

  if (medianViewsPerHour > 0 && viewsPerHour < medianViewsPerHour * 0.35) {
    labels.push("low_exposure_proxy");
  }

  if (retention !== null && medianRetention !== null && retention < medianRetention * 0.75) {
    labels.push("weak_hold");
  }

  if (
    retention !== null &&
    medianRetention !== null &&
    retention >= medianRetention * 1.1 &&
    medianViews > 0 &&
    views < medianViews * 0.6
  ) {
    labels.push("strong_retention_low_distribution");
  }

  if (
    likeRate !== null &&
    medianLikeRate !== null &&
    likeRate >= medianLikeRate * 1.25 &&
    medianViews > 0 &&
    views < medianViews * 0.7
  ) {
    labels.push("high_like_rate_low_scale");
  }

  if (
    medianViews > 0 &&
    views >= medianViews * 1.2 &&
    retention !== null &&
    medianRetention !== null &&
    retention >= medianRetention
  ) {
    labels.push("winner_candidate");
  }

  const uniqueLabels = [...new Set(labels.length > 0 ? labels : ["mid_video_drop_proxy"])];
  const priority = uniqueLabels.includes("weak_hold") ? "high" : uniqueLabels.includes("winner_candidate") ? "low" : "medium";

    suggestionsEn: {
      labels: uniqueLabels,

function stableJson(value: unknown): string {
  if (Array.isArray(value)) {
    return `[${value.map(stableJson).join(",")}]`;
  }

  if (typeof value === "object" && value !== null) {
    return `{${Object.entries(value as Record<string, unknown>)
      .sort(([a], [b]) => a.localeCompare(b))
      .map(([key, nested]) => `${JSON.stringify(key)}:${stableJson(nested)}`)
      .join(",")}}`;
  }

  return JSON.stringify(value);
}

export function buildYoutubeDiagnosisInputHash(input: YoutubeDiagnosisInput) {
  return `sha256:${createHash("sha256").update(stableJson(input)).digest("hex")}`;
}

export function parseYoutubeDiagnosisOutput(value: string): YoutubeDiagnosisResponse {
  return youtubeDiagnosisResponseSchema.parse(JSON.parse(value));
}

function extractResponseText(response: OpenAI.Responses.Response): string {
  if (typeof response.output_text === "string" && response.output_text.trim()) {
    return response.output_text.trim();
  }

  const chunks: string[] = [];
  for (const output of response.output ?? []) {
    if (output.type !== "message") continue;
    for (const content of output.content ?? []) {
      if (content.type === "output_text") {
        chunks.push(content.text);
      }
    }
  }

  const text = chunks.join("").trim();
  if (!text) {
    throw new Error("youtube_diagnosis_response_invalid");
  }

  return text;
}

export async function diagnoseYoutubeVideo(input: {
  diagnosisInput: YoutubeDiagnosisInput;
  client?: OpenAI;

  const client = input.client ?? new OpenAI({ apiKey });

  if (!apiKey && !input.client) {
    throw new Error("OPENAI_API_KEY_missing");
  }

  const response = await client.responses.create({
    model,
    input: [
      {
        role: "developer",
        content:
          "You diagnose YouTube Shorts performance for Tiny Mechanisms. Return Thai analysis fields and English rewrite suggestions only. Be concrete, conservative, and do not recommend automatic publishing or unsafe claims.",
      },
      {
        role: "user",
        content: JSON.stringify(input.diagnosisInput),
      },
    ],
    reasoning: model === "gpt-5.5" ? { effort: YOUTUBE_DIAGNOSIS_REASONING_EFFORT } : undefined,
    text: {
      format: {
        type: "json_schema",
        name: "youtube_video_diagnosis_v1",
        schema: YOUTUBE_DIAGNOSIS_JSON_SCHEMA as unknown as Record<string, unknown>,
        strict: true,
      },
    },
  });

  const parsed = parseYoutubeDiagnosisOutput(extractResponseText(response));

  return {
    diagnosis: parsed,
    model: response.model ?? model,
    reasoningEffort: model === "gpt-5.5" ? YOUTUBE_DIAGNOSIS_REASONING_EFFORT : null,
    responseId: response.id,
    status: response.status,
  };
}

  youtubeAnalyticsRefreshRequestSchema,
  type YoutubeAiDiagnosisResponse,
  type YoutubeAnalyticsDashboardResponse,
} from "@short-workflow/shared";
import type { DbClient } from "@short-workflow/db";
import { Elysia } from "elysia";

import { conflict, internalError, jsonError, validationFailed } from "../http";

type StatusSetter = {
  status?: number | string;
};

type RouteContext = {
  db: DbClient;
  set: StatusSetter;
  body: unknown;
  query: { windowDays?: string };
  params: { youtubeVideoId?: string };
};

function withRouteContext(context: unknown) {
  return context as RouteContext;
}

export type AnalyticsRouteServices = {
  getDashboard: (
    db: DbClient,
    input: { windowDays: number },
  ) => Promise<YoutubeAnalyticsDashboardResponse>;
  refreshDashboard: (
    db: DbClient,
    input: { windowDays: number },
  ) => Promise<YoutubeAnalyticsDashboardResponse>;
  analyzeVideo: (
    db: DbClient,
    input: { youtubeVideoId: string },
  ) => Promise<YoutubeAiDiagnosisResponse>;
};

function unavailableServices(): AnalyticsRouteServices {
  const unavailable = async () => {
    throw new Error("analytics_services_not_configured");
  };

  return {
    getDashboard: unavailable,
    refreshDashboard: unavailable,
    analyzeVideo: unavailable,
  };
}

function mapAnalyticsError(set: StatusSetter, error: unknown) {
  if (!(error instanceof Error)) {
    throw error;
  }

  if (error.message === "youtube_not_connected" || error.message === "youtube_reconnect_required") {
    return conflict(set, error.message);
  }

  if (error.message === "youtube_video_not_found") {
    return jsonError(set, 404, "youtube_video_not_found");
  }

  if (error.message === "youtube_analytics_snapshot_missing") {
    return jsonError(set, 422, "youtube_analytics_snapshot_missing");
  }

  if (
    error.message.startsWith("youtube_analytics_fetch_failed") ||
    error.message === "youtube_ai_diagnosis_failed"
  ) {
    return jsonError(set, 502, error.message);
  }

  throw error;
}

export function createAnalyticsRoutes(services: AnalyticsRouteServices = unavailableServices()) {
  return new Elysia({ prefix: "/analytics" })
    .get("/youtube", async (context) => {
      const { db, query, set } = withRouteContext(context);
      const parsed = youtubeAnalyticsRefreshRequestSchema.safeParse({
        windowDays: query.windowDays === undefined ? undefined : Number(query.windowDays),
      });
      if (!parsed.success) {
        return validationFailed(set, parsed.error);
      }

      try {
        return await services.getDashboard(db, parsed.data);
      } catch (error) {
        return mapAnalyticsError(set, error);
      }
    })
    .post("/youtube/refresh", async (context) => {
      const { body, db, set } = withRouteContext(context);
      const parsed = youtubeAnalyticsRefreshRequestSchema.safeParse(body ?? {});
      if (!parsed.success) {
        return validationFailed(set, parsed.error);
      }

      try {
        return await services.refreshDashboard(db, parsed.data);
      } catch (error) {
        return mapAnalyticsError(set, error);
      }
    })
    .post("/youtube/videos/:youtubeVideoId/analyze", async (context) => {
      const { db, params, set } = withRouteContext(context);
      const parsed = youtubeAiDiagnosisRequestSchema.safeParse({
        youtubeVideoId: params.youtubeVideoId,
      });
      if (!parsed.success) {
        return validationFailed(set, parsed.error);
      }

      try {
        return await services.analyzeVideo(db, parsed.data);
      } catch (error) {
        return mapAnalyticsError(set, error);
      }
    })
    .onError(({ set }) => internalError(set));
}

  const numeric = values.filter((value): value is number => typeof value === "number").sort((a, b) => a - b);

  videos: YoutubeAnalyticsVideoSummary[];

  const totalViews = input.videos.reduce(
    (sum, video) => sum + (video.latestSnapshot?.views ?? 0),
    0,
  );
  const best = [...input.videos].sort(
    (a, b) => (b.latestSnapshot?.views ?? 0) - (a.latestSnapshot?.views ?? 0),
  )[0];
  const needsAttentionCount = input.videos.filter((video) => {
    const suggestions = video.latestRuleDiagnosis?.suggestionsEn as { labels?: unknown } | undefined;
    return Array.isArray(suggestions?.labels)
      ? suggestions.labels.some((label) => label === "weak_hold" || label === "low_exposure_proxy")
      : false;
  }).length;

  return {
    auth: input.auth,
    windowDays: input.windowDays,
    aggregates: {
      recentVideoCount: input.videos.length,
      totalViews,
      medianAverageViewPercentage: median(
        input.videos.map((video) => video.latestSnapshot?.averageViewPercentage),
      ),
      needsAttentionCount,
      bestPerformerVideoId: best?.latestSnapshot?.views ? best.link.youtubeVideoId : null,
    },
    videos: input.videos,
  };
}

async function loadDashboard(db: DbClient, input: { windowDays: number }) {
  const auth = await getYoutubeAuthStatus();
  const since = new Date(Date.now() - input.windowDays * 24 * 60 * 60 * 1000);
  const links = await listRecentYoutubeVideoLinks(db, since);
  const linkIds = links.map((link) => link.id);
  const snapshots = await listLatestYoutubeAnalyticsSnapshots(db, linkIds);
  const diagnoses = await listLatestYoutubeVideoDiagnoses(db, { youtubeVideoLinkIds: linkIds });
  const creativeContexts = await Promise.all(
    links.map((link) => getYoutubeCreativeContext(db, link.youtubeVideoId)),
  );

  const videos = links.map((link, index) => {
    const latestSnapshot =
      snapshots.find((snapshot) => snapshot.youtubeVideoLinkId === link.id) ?? null;
    const latestRuleDiagnosis =
      diagnoses.find(
        (diagnosis) =>
          diagnosis.youtubeVideoLinkId === link.id && diagnosis.diagnosisType === "rule_based",
      ) ?? null;
    const latestAiDiagnosis =
      diagnoses.find(
        (diagnosis) =>
          diagnosis.youtubeVideoLinkId === link.id && diagnosis.diagnosisType === "ai",
      ) ?? null;

    return {
      link: toApiVideoLink(link),
      latestSnapshot: latestSnapshot ? toApiSnapshot(latestSnapshot) : null,
      latestRuleDiagnosis: latestRuleDiagnosis ? toApiDiagnosis(latestRuleDiagnosis) : null,
      latestAiDiagnosis: latestAiDiagnosis ? toApiDiagnosis(latestAiDiagnosis) : null,
      creativeContext: creativeContexts[index] ?? null,
    };
  });

  return buildAnalyticsDashboard({
    auth,
    windowDays: input.windowDays,
    videos,
  });

async function refreshYoutubeAnalyticsDashboard(db: DbClient, input: { windowDays: number }) {
  const now = new Date();
  const accessToken = await readFreshYoutubeAccessToken();
  const recentIds = await fetchRecentChannelVideos({
    accessToken,
    windowDays: input.windowDays,
    now,
  });
  const details = await fetchYoutubeVideoDetails({
    accessToken,
    youtubeVideoIds: recentIds,
  });
  const analyticsRows = analyticsByVideoId(
    await fetchYoutubeAnalyticsRows({
      accessToken,
      youtubeVideoIds: recentIds,
      startDate: dateOnly(new Date(now.getTime() - input.windowDays * 24 * 60 * 60 * 1000)),
      endDate: dateOnly(now),
    }),
  );
  const mappings = await listYoutubeUploadVideoMappings(db);
  const mappingByVideoId = new Map(
    mappings
      .filter((mapping) => mapping.youtubeVideoId)
      .map((mapping) => [mapping.youtubeVideoId, mapping]),
  );

  const createdSnapshots: YoutubeAnalyticsSnapshotRow[] = [];
  for (const item of details) {
    const mapping = mappingByVideoId.get(item.id);
    const publishedAt = parseDate(item.snippet?.publishedAt);
    const analytics = analyticsRows.get(item.id);
    const views =
      metric(analytics, "views") ??
      normalizeYoutubeMetricNumber(item.statistics?.viewCount);
    const likes =
      metric(analytics, "likes") ??
      normalizeYoutubeMetricNumber(item.statistics?.likeCount);
    const comments =
      metric(analytics, "comments") ??
      normalizeYoutubeMetricNumber(item.statistics?.commentCount);
    const derived = deriveSnapshotMetrics({ views, likes, publishedAt, now });
    const link = await upsertYoutubeVideoLink(db, {
      youtubeVideoId: item.id,
      projectId: mapping?.projectId ?? null,
      uploadJobId: mapping?.uploadJobId ?? null,
      source: mapping ? "db_upload" : "channel_discovery",
      linkStatus: mapping ? "linked" : "unlinked",
      title: item.snippet?.title?.trim() || "Untitled YouTube video",
      description: item.snippet?.description ?? null,
      publishedAt,
      durationSeconds: item.contentDetails?.duration
        ? parseIso8601DurationSeconds(item.contentDetails.duration)
        : null,
      privacyStatus: item.status?.privacyStatus ?? null,
      lastSyncedAt: now,
    });

    createdSnapshots.push(
      await createYoutubeAnalyticsSnapshot(db, {
        youtubeVideoLinkId: link.id,
        youtubeVideoId: item.id,
        windowDays: input.windowDays,
        views,
        engagedViews: metric(analytics, "engagedViews"),
        likes,
        comments,
        shares: metric(analytics, "shares"),
        subscribersGained: metric(analytics, "subscribersGained"),
        averageViewDurationSeconds: metric(analytics, "averageViewDuration"),
        averageViewPercentage: metric(analytics, "averageViewPercentage"),
        viewsPerHour: derived.viewsPerHour,
        likeRate: derived.likeRate,
        rawDataApi: item as Record<string, unknown>,
        rawAnalyticsApi: analytics ?? {},
      }),
    );
  }

  const recentMedians = {
    views: median(createdSnapshots.map((snapshot) => snapshot.views)),
    viewsPerHour: median(createdSnapshots.map((snapshot) => snapshot.viewsPerHour)),
    averageViewPercentage: median(
      createdSnapshots.map((snapshot) => snapshot.averageViewPercentage),
    ),
    likeRate: median(createdSnapshots.map((snapshot) => snapshot.likeRate)),
  };

  for (const snapshot of createdSnapshots) {
    const link = await getYoutubeVideoLinkByVideoId(db, snapshot.youtubeVideoId);
    if (!link) continue;
    const ageHours =
      link.publishedAt === null
        ? null
        : Math.max(0, (now.getTime() - link.publishedAt.getTime()) / 3_600_000);
    const rule = buildRuleDiagnosis({
      snapshot,
      recentMedians,
      ageHours,
    });
    const inputHash = buildYoutubeDiagnosisInputHash({
      type: "rule_based",
      youtubeVideoId: snapshot.youtubeVideoId,
      snapshotId: snapshot.id,
      labels: rule.labels,
      metrics: {
        views: snapshot.views,
        viewsPerHour: snapshot.viewsPerHour,
        averageViewPercentage: snapshot.averageViewPercentage,
        likeRate: snapshot.likeRate,
      },
    });

    await upsertYoutubeVideoDiagnosis(db, {
      youtubeVideoLinkId: link.id,
      snapshotId: snapshot.id,
      diagnosisType: "rule_based",
      model: null,
      reasoningEffort: null,
      inputHash,
      summaryTh: rule.summaryTh,
      suggestionsEn: rule.suggestionsEn,
      rawOutput: rule,
    });
  }

  return loadDashboard(db, input);
}

async function analyzeYoutubeVideoWithAi(db: DbClient, input: { youtubeVideoId: string }) {
  const link = await getYoutubeVideoLinkByVideoId(db, input.youtubeVideoId);
  if (!link) {
    throw new Error("youtube_video_not_found");
  }

  const [snapshot] = await listLatestYoutubeAnalyticsSnapshots(db, [link.id]);
  if (!snapshot) {
    throw new Error("youtube_analytics_snapshot_missing");
  }

  const [latestRuleDiagnosis] = await listLatestYoutubeVideoDiagnoses(db, {
    youtubeVideoLinkIds: [link.id],
    diagnosisType: "rule_based",
  });
  const creativeContext = await getYoutubeCreativeContext(db, input.youtubeVideoId);
  const diagnosisInput = {
    link: toApiVideoLink(link),
    latestSnapshot: toApiSnapshot(snapshot),
    latestRuleDiagnosis: latestRuleDiagnosis ? toApiDiagnosis(latestRuleDiagnosis) : null,
    creativeContext,
  };
  const inputHash = buildYoutubeDiagnosisInputHash(diagnosisInput);
  const ai = await diagnoseYoutubeVideo({ diagnosisInput });
  const row = await upsertYoutubeVideoDiagnosis(db, {
    youtubeVideoLinkId: link.id,
    snapshotId: snapshot.id,
    diagnosisType: "ai",
    model: ai.model,
    reasoningEffort: ai.reasoningEffort,
    inputHash,
    summaryTh: ai.diagnosis.summaryTh,
    suggestionsEn: ai.diagnosis,
    rawOutput: {
      responseId: ai.responseId,
      status: ai.status,
      diagnosis: ai.diagnosis,
    },
  });

  return { diagnosis: toApiDiagnosis(row) };
}

export function formatMetric(value: number | null | undefined) {
  if (value === null || value === undefined) {
    return "-";
  }

  return new Intl.NumberFormat(undefined, {
    maximumFractionDigits: Number.isInteger(value) ? 0 : 1,
  }).format(value);
}

export function formatPercent(value: number | null | undefined) {
  if (value === null || value === undefined) {
    return "-";
  }

  return `${new Intl.NumberFormat(undefined, {
    maximumFractionDigits: 1,
  }).format(value)}%`;
}

export function formatDateTime(value: string | null | undefined) {
  if (!value) {
    return "-";
  }

  return new Intl.DateTimeFormat(undefined, {
    dateStyle: "medium",
    timeStyle: "short",
  }).format(new Date(value));
}

export function statusLabel(value: "linked" | "unlinked") {
  return value === "linked" ? "Linked" : "Unlinked";
}

function errorCode(error: unknown) {
  const payload = (error as ApiError | undefined)?.payload;
  return typeof payload === "object" && payload !== null && "error" in payload
    ? String((payload as { error: unknown }).error)
    : null;
}

function SkeletonRows() {

export function AnalyticsDashboard() {
  const windowDays = 30;
  const [selectedVideoId, setSelectedVideoId] = useState<string | null>(null);
  const dashboardQuery = useYoutubeAnalyticsQuery(windowDays);
  const refreshMutation = useRefreshYoutubeAnalyticsMutation(windowDays);
  const analyzeMutation = useAnalyzeYoutubeVideoMutation(windowDays);
  const dashboard = dashboardQuery.data;
  const selectedVideo = useMemo(() => {
    if (!dashboard?.videos.length) {
      return null;
    }

    return (
      dashboard.videos.find((video) => video.link.youtubeVideoId === selectedVideoId) ??
      dashboard.videos[0] ??
      null
    );
  }, [dashboard?.videos, selectedVideoId]);
  const refreshError = errorCode(refreshMutation.error);

  return (
    <div className="space-y-5">
      <header className="flex flex-col gap-4 border-b border-border pb-5 lg:flex-row lg:items-end lg:justify-between">
        <div>
          <p className="text-sm font-medium text-muted-foreground">YouTube</p>
          <h1 className="text-2xl font-semibold tracking-normal">Analytics</h1>
          <p className="mt-1 max-w-2xl text-sm text-muted-foreground">
            Recent Shorts performance from YouTube APIs, linked with local project context when available.
          </p>
        </div>
        <button
          className="inline-flex h-10 items-center justify-center gap-2 rounded-md bg-primary px-4 text-sm font-medium text-primary-foreground transition hover:bg-primary/90 active:translate-y-px disabled:cursor-not-allowed disabled:opacity-60"
          disabled={refreshMutation.isPending}
          onClick={() => refreshMutation.mutate({ windowDays })}
          type="button"
        >
          <RefreshCw
            className={refreshMutation.isPending ? "size-4 animate-spin" : "size-4"}
            aria-hidden="true"
          />
          Refresh Analytics
        </button>
      </header>

      {dashboard?.auth.reconnectRequired || refreshError === "youtube_reconnect_required" ? (
        <div className="flex items-start gap-2 border border-accent/30 bg-accent/10 p-3 text-sm text-accent-foreground">
          <AlertCircle className="mt-0.5 size-4" aria-hidden="true" />
          Reconnect YouTube to grant analytics read scopes.
        </div>
      ) : null}

      <section className="grid gap-3 border-y border-border py-4 sm:grid-cols-2 lg:grid-cols-5">
        <Metric label="Recent videos" value={formatMetric(dashboard?.aggregates.recentVideoCount)} />
        <Metric label="Total views" value={formatMetric(dashboard?.aggregates.totalViews)} />
        <Metric
          label="Median viewed"
          value={formatPercent(dashboard?.aggregates.medianAverageViewPercentage)}
        />
        <Metric label="Needs attention" value={formatMetric(dashboard?.aggregates.needsAttentionCount)} />
        <Metric label="Best performer" value={dashboard?.aggregates.bestPerformerVideoId ?? "-"} />
      </section>

      {dashboardQuery.isLoading ? (
        <SkeletonRows />
      ) : dashboardQuery.error ? (
        <div className="border border-accent/30 bg-accent/10 p-4 text-sm text-accent-foreground">
          Analytics could not be loaded.
        </div>

        <div className="border-y border-border bg-card p-6">
          <div className="flex items-center gap-2 text-sm font-medium">
            <BarChart3 className="size-4 text-muted-foreground" aria-hidden="true" />
            No analytics snapshots yet
          </div>
          <p className="mt-1 text-sm text-muted-foreground">
            No snapshots for the selected window.
          </p>
        </div>
      ) : dashboard ? (
        <div className="grid gap-5 lg:grid-cols-[minmax(0,1fr)_360px]">
          <AnalyticsTable
            analyzingVideoId={analyzeMutation.isPending ? analyzeMutation.variables ?? null : null}
            onAnalyze={(youtubeVideoId) => analyzeMutation.mutate(youtubeVideoId)}
            onSelect={setSelectedVideoId}
            selectedVideoId={selectedVideo?.link.youtubeVideoId ?? null}
            videos={dashboard.videos}
          />
          <AnalyticsDetailPanel video={selectedVideo} />
        </div>
      ) : null}
    </div>
  );
}
